erDiagram
    USER ||--o{ TRADE : executes
    TRADE ||--o{ FIFOLOT : creates_or_consumes
    TRADE ||--|| GAIN_RECORD : generates
    TRADE }o--|| REFERENCE_DATA : uses
    FIFOLOT }o--|| REFERENCE_DATA : indexed_by

    USER {
        int user_id PK
        string name
        string tax_status "Full Taxpayer"
    }

    TRADE {
        int trade_id PK
        int user_id FK
        string ticker
        string type "BUY / SELL"
        float quantity
        float price_fc "Foreign Currency"
        date trade_date
        float cbrt_rate
    }

    FIFOLOT {
        int lot_id PK
        int purchase_trade_id FK
        float original_qty
        float remaining_qty
        float cost_basis_tl
        date purchase_date
    }

    GAIN_RECORD {
        int gain_id PK
        int sell_trade_id FK
        float taxable_income_tl
        boolean is_indexed
        string tax_bracket
    }

    REFERENCE_DATA {
        int ref_id PK
        date entry_date
        float cbrt_buying_rate
        float wpi_index_value "Yİ-ÜFE"
    }
