    USER ||--o{ TRADE : executes
    TRADE ||--o{ FIFOLOT : creates_or_consumes
    TRADE ||--o{ GAIN_RECORD : generates
    TRADE }o--|| REFERENCE_DATA : uses
    FIFOLOT }o--|| REFERENCE_DATA : indexed_by

    USER {
        uuid user_id PK
        string name
        string tax_status "Full Taxpayer"
    }

    TRADE {
        uuid trade_id PK
        uuid user_id FK
        string ticker
        string trade_type "BUY / SELL"
        decimal quantity
        decimal price_fc "Foreign Currency"
        date trade_date
        string currency
    }

    FIFOLOT {
        uuid lot_id PK
        uuid purchase_trade_id FK
        string ticker
        decimal original_qty
        decimal remaining_qty
        decimal cost_basis_tl
        date purchase_date
    }

    GAIN_RECORD {
        uuid gain_id PK
        uuid sell_trade_id FK
        string ticker
        decimal matched_qty
        decimal realized_gain_tl
        decimal wpi_ratio
        boolean is_indexed
        string audit_note
    }

    REFERENCE_DATA {
        int ref_id PK
        date entry_date
        decimal cbrt_buying_rate

    }
